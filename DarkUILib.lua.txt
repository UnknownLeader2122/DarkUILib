– DarkUI Library – Simple dark-mode UI with Tabs, Buttons, Toggles

local Library = {} Library.__index = Library

local theme = { Background = Color3.fromRGB(20, 20, 20), Sidebar =
Color3.fromRGB(15, 15, 15), Tab = Color3.fromRGB(35, 35, 35), Accent =
Color3.fromRGB(80, 140, 255), Text = Color3.fromRGB(230, 230, 230),
ToggleOff = Color3.fromRGB(90, 90, 90), ToggleOn = Color3.fromRGB(0,
200, 130) }

local function Make(class, props) local obj = Instance.new(class) for
k,v in pairs(props) do obj[k] = v end return obj end

function Library:CreateWindow(title) local gui = Make(“ScreenGui”, {
Parent = game:GetService(“CoreGui”), ResetOnSpawn = false })

    local main = Make("Frame", {
        Parent = gui,
        Size = UDim2.new(0, 500, 0, 300),
        Position = UDim2.new(0.5, -250, 0.5, -150),
        BackgroundColor3 = theme.Background
    })

    Make("UICorner", {Parent = main})

    local titleBar = Make("TextLabel", {
        Parent = main,
        Text = title,
        Size = UDim2.new(1, 0, 0, 35),
        BackgroundColor3 = theme.Sidebar,
        TextColor3 = theme.Text,
        Font = Enum.Font.GothamSemibold,
        TextSize = 16
    })

    local sidebar = Make("Frame", {
        Parent = main,
        Size = UDim2.new(0, 120, 1, -35),
        Position = UDim2.new(0, 0, 0, 35),
        BackgroundColor3 = theme.Sidebar
    })

    local tabHolder = Make("Frame", {
        Parent = main,
        Size = UDim2.new(1, -120, 1, -35),
        Position = UDim2.new(0, 120, 0, 35),
        BackgroundColor3 = theme.Background
    })

    local layout = Make("UIListLayout", {
        Parent = sidebar,
        Padding = UDim.new(0, 5)
    })

    local window = {
        gui = gui,
        main = main,
        tabs = {},
        tabHolder = tabHolder,
        sidebar = sidebar,
    }

    setmetatable(window, Library)
    return window

end

function Library:AddTab(name) local tabButton = Make(“TextButton”, {
Parent = self.sidebar, Text = name, Size = UDim2.new(1, -10, 0, 30),
BackgroundColor3 = theme.Tab, TextColor3 = theme.Text, Font =
Enum.Font.Gotham, TextSize = 14 })

    Make("UICorner", {Parent = tabButton})

    local tabFrame = Make("ScrollingFrame", {
        Parent = self.tabHolder,
        Visible = false,
        CanvasSize = UDim2.new(0,0,0,0),
        BackgroundTransparency = 1,
        Size = UDim2.new(1,0,1,0)
    })

    local layout = Make("UIListLayout", {
        Parent = tabFrame,
        Padding = UDim.new(0, 6)
    })

    local tab = { frame = tabFrame }

    function tab:Show()
        for _,t in ipairs(self.Parent.tabs) do
            t.frame.Visible = false
        end
        self.frame.Visible = true
    end

    tabButton.MouseButton1Click:Connect(function()
        tab:Show()
    end)

    table.insert(self.tabs, tab)
    tab.Parent = self

    return tab

end

function Library:AddButton(name, callback)
    local btn = Make("TextButton", {
        Parent = self.frame, -- only works if self.frame exists
        Text = name,
        Size = UDim2.new(0, 300, 0, 35),
        BackgroundColor3 = theme.Tab,
        TextColor3 = theme.Text,
        Font = Enum.Font.Gotham,
        TextSize = 14,
    })

    Make("UICorner", {Parent = btn})

    btn.MouseButton1Click:Connect(callback)
end

function Library:AddToggle(name, default, callback)
    local holder = Make("Frame", {
        Parent = self.frame, -- again, must exist
        Size = UDim2.new(0, 300, 0, 35),
        BackgroundTransparency = 1,
    })

    local label = Make("TextLabel", {
        Parent = holder,
        Text = name,
        Size = UDim2.new(1, -40, 1, 0),
        BackgroundTransparency = 1,
        TextColor3 = theme.Text,
        Font = Enum.Font.Gotham,
        TextSize = 14,
        TextXAlignment = Enum.TextXAlignment.Left
    })

    local toggle = Make("TextButton", {
        Parent = holder,
        Size = UDim2.new(0, 30, 0, 18),
        Position = UDim2.new(1, -30, 0.5, -9),
        BackgroundColor3 = default and theme.ToggleOn or theme.ToggleOff,
        Text = "",
    })
    Make("UICorner", {Parent = toggle})

    local state = default
    toggle.MouseButton1Click:Connect(function()
        state = not state
        toggle.BackgroundColor3 = state and theme.ToggleOn or theme.ToggleOff
        callback(state)
    end)
end

return Library
